}

func (s ConcreteStrategy2) Execute() {
	fmt.Println("concrete strategy2 execute")
}

type Context struct {
	strategy Strategy
}

func (c *Context) SetStrategy(strategy Strategy) {
	c.strategy = strategy
}

func (c *Context) Do() {
	fmt.Println("do something")
	c.strategy.Execute()
}

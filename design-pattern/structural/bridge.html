	fmt.Println("ConcreateImplA")

	fmt.Println("ConcreateImplB")

	Operation()

	Impl Implementor
}

	fmt.Println("RefineAbstractionA")
	ra.Impl.OperationImpl()
}

	Impl Implementor
}

	fmt.Println("RefineAbstractionB")
	ra.Impl.OperationImpl()

	if atomic.LoadUint32(&flag) == 0 {
		rwlock.Lock()
		defer rwlock.Unlock()
		if atomic.LoadUint32(&flag) == 0 {

			fmt.Println("Single Instance already created-1")
		}

		fmt.Println("Single Instance already created-2")
	}
	return singleInstance4
}
